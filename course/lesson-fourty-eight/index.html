Capstone Project: Enhanced E-commerce App

Welcome to Lesson 48, the final lesson of CODEXEDOC’s Full-Stack Development Course! In this capstone project, you’ll enhance your e-commerce app by integrating all skills from Lessons 1–47, adding features like search, checkout, and user profiles, and deploying the app.

What is the Capstone Project?

The capstone combines frontend (React, Redux), backend (Express, MongoDB), real-time features (WebSockets), optimization, and CI/CD into a production-ready e-commerce app.

Why Build This Project?

This project showcases your full-stack skills, creating a portfolio piece with advanced features like search and checkout, ready for real-world use.

Key Project Features

Implement these features:

  • Search: Filter products by name.
  • Checkout: Process orders with a mock payment.
  • User Profiles: Display order history.
  • Deployment: Use Vercel and Render with CI/CD.

Building the Project

Follow these steps:

  1. Add Search: Implement frontend and backend search.
  2. Build Checkout: Create a checkout flow.
  3. Create User Profiles: Store and display orders.
  4. Deploy: Use Lesson 47’s CI/CD pipeline.

const express = require('express');
const mongoose = require('mongoose');
const cors = require('cors');
const jwt = require('jsonwebtoken');
const { Server } = require('socket.io');
const http = require('http');

const app = express();
const server = http.createServer(app);
const io = new Server(server, { cors: { origin: 'http://localhost:5173' } });

app.use(cors());
app.use(express.json());

mongoose.connect('mongodb://localhost:27017/ecommerce', {
    useNewUrlParser: true,
    useUnifiedTopology: true
});

const userSchema = new mongoose.Schema({
    email: String,
    password: String,
    orders: [{ type: mongoose.Schema.Types.ObjectId, ref: 'Order' }]
});
const productSchema = new mongoose.Schema({
    name: String,
    price: Number,
    image: String
});
const orderSchema = new mongoose.Schema({
    userId: { type: mongoose.Schema.Types.ObjectId, ref: 'User' },
    items: [{ productId: String, quantity: Number, price: Number }],
    total: Number,
    date: { type: Date, default: Date.now }
});

const User = mongoose.model('User', userSchema);
const Product = mongoose.model('Product', productSchema);
const Order = mongoose.model('Order', orderSchema);

app.post('/api/auth/login', async (req, res) => {
    const { email, password } = req.body;
    const user = await User.findOne({ email, password });
    if (!user) return res.status(401).json({ error: 'Invalid credentials' });
    const token = jwt.sign({ userId: user._id }, 'secret', { expiresIn: '1h' });
    res.json({ token });
});

app.get('/api/products', async (req, res) => {
    const { search } = req.query;
    const query = search ? { name: { $regex: search, $options: 'i' } } : {};
    const products = await Product.find(query);
    res.json(products);
});

app.post('/api/orders', async (req, res) => {
    const { items, total } = req.body;
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: 'Unauthorized' });
    const { userId } = jwt.verify(token, 'secret');
    const order = new Order({ userId, items, total });
    await order.save();
    await User.findByIdAndUpdate(userId, { $push: { orders: order._id } });
    io.emit('newOrder', order);
    res.status(201).json(order);
});

app.get('/api/profile', async (req, res) => {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: 'Unauthorized' });
    const { userId } = jwt.verify(token, 'secret');
    const user = await User.findById(userId).populate('orders');
    res.json(user);
});

server.listen(3000, () => console.log('Server running'));

import { useState } from 'react';

function Checkout({ cart }) {
    const [message, setMessage] = useState('');

    const handleCheckout = async () => {
        const token = localStorage.getItem('token');
        if (!token) {
            setMessage('Please login to checkout');
            return;
        }
        try {
            const items = cart.map(item => ({
                productId: item.id,
                quantity: item.quantity,
                price: item.price
            }));
            const total = cart.reduce((sum, item) => sum + item.price * item.quantity, 0);
            const res = await fetch('http://localhost:3000/api/orders', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': `Bearer ${token}`
                },
                body: JSON.stringify({ items, total })
            });
            if (!res.ok) throw new Error('Checkout failed');
            setMessage('Order placed successfully!');
        } catch (err) {
            setMessage('Checkout error: ' + err.message);
        }
    };

    return (

{message}
}
        

    );
}

export default Checkout;

Advanced API Integration in React

Welcome to Lesson 44 of CODEXEDOC’s Full-Stack Development Course! Advanced API integration enhances your React app with secure, authenticated requests. In this lesson, you’ll connect your React frontend to Lesson 40’s backend, adding login and protected routes.

What is Advanced API Integration?

Advanced integration involves handling authenticated API calls, token storage, and error management in React, ensuring secure data flow between frontend and backend.

Why Integrate APIs?

Integration enables features like user login and secure product management in your e-commerce app, leveraging Lesson 35’s JWT authentication.

Key Integration Concepts

Focus on these techniques:

  • Authenticated Requests: Include JWT in headers.
  • Token Storage: Use localStorage for persistence.
  • Error Handling: Manage 401/403 errors gracefully.
  • Async Actions: Use Redux Toolkit for API calls.

Implementing Integration

Follow these steps:

  1. Add Login Component: Create a login form.
  2. Store Tokens: Save JWT in localStorage.
  3. Fetch with Auth: Include tokens in API calls.
  4. Test Integration: Verify login and protected routes.

Code Example: API Integration

Here’s a React frontend with login and authenticated API calls:

<!-- src/App.jsx -->
import { useState } from 'react';
import { useSelector, useDispatch } from 'react-redux';
import { addToCart, removeFromCart } from './cartSlice';
import Login from './Login';

function App() {
    const [products, setProducts] = useState([]);
    const [error, setError] = useState(null);
    const cart = useSelector(state => state.cart);
    const dispatch = useDispatch();

    const fetchProducts = async () => {
        try {
            const token = localStorage.getItem('token');
            const res = await fetch('http://localhost:3000/api/products', {
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': token ? `Bearer ${token}` : ''
                }
            });
            if (!res.ok) throw new Error('Failed to fetch');
            const data = await res.json();
            setProducts(data);
        } catch (err) {
            setError(err.message);
        }
    };

    return (
        <div className="p-4">
            <header className="bg-blue-600 text-white p-4">
                <div className="flex justify-between">
                    <h1 className="text-2xl font-bold">Store</h1>
                    <Login fetchProducts={fetchProducts} />
                </div>
            </header>
            <main className="container mx-auto">
                {error && <p className="text-red-600">Error: {error}</p>}
                <h2 className="text-xl font-bold my-4">Products</h2>
                <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                    {products.map(product => (
                        <div key={product.id} className="bg-white p-4 rounded shadow">
                            <h3 className="font-bold">{product.name}</h3>
                            <p>${product.price.toFixed(2)}</p>
                            <button
                                className="bg-blue-600 text-white px-4 py-2 rounded"
                                onClick={() => dispatch(addToCart(product))}
                            >
                                Add to Cart
                            </button>
                        </div>
                    ))}
                </div>
                <h2 className="text-xl font-bold my-4">Cart</h2>
                <ul className="bg-white p-4 rounded shadow">
                    {cart.map(item => (
                        <li key={item.id} className="flex justify-between py-2">
                            <span>{item.name} (x{item.quantity}) - ${(item.price * item.quantity).toFixed(2)}</span>
                            <button
                                className="text-red-600"
                                onClick={() => dispatch(removeFromCart(item.id))}
                            >
                                Remove
                            </button>
                        </li>
                    ))}
                </ul>
                <p className="mt-4 font-semibold">
                    Total: ${cart.reduce((sum, item) => sum + item.price * item.quantity, 0).toFixed(2)}
                </p>
            </main>
        </div>
    );
}

export default App;

<!-- src/Login.jsx -->
import { useState } from 'react';

function Login({ fetchProducts }) {
    const [email, setEmail] = useState('');
    const [password, setPassword] = useState('');
    const [isLoggedIn, setIsLoggedIn] = useState(!!localStorage.getItem('token'));

    const handleLogin = async () => {
        try {
            const res = await fetch('http://localhost:3000/api/auth/login', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ email, password })
            });
            if (!res.ok) throw new Error('Login failed');
            const data = await res.json();
            localStorage.setItem('token', data.token);
            setIsLoggedIn(true);
            fetchProducts();
        } catch (err) {
            alert('Login error: ' + err.message);
        }
    };

    const handleLogout = () => {
        localStorage.removeItem('token');
        setIsLoggedIn(false);
    };

    return (
        <div>
            {isLoggedIn ? (
                <button
                    className="bg-white text-blue-600 px-4 py-2 rounded"
                    onClick={handleLogout}
                >
                    Logout
                </button>
            ) : (
                <div>
                    <input
                        type="email"
                        value={email}
                        onChange={e => setEmail(e.target.value)}
                        placeholder="Email"
                        className="border p-2 mr-2"
                    />
                    <input
                        type="password"
                        value={password}
                        onChange={e => setPassword(e.target.value)}
                        placeholder="Password"
                        className="border p-2 mr-2"
                    />
                    <button
                        className="bg-white text-blue-600 px-4 py-2 rounded"
                        onClick={handleLogin}
                    >
                        Login
                    </button>
                </div>
            )}
        </div>
    );
}

export default Login;
